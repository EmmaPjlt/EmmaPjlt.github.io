canvas 实用方法

canvas与绘制页面不同，不支持设置css属性，不支持文字自动换行，不支持元素自动垂直定位，所有都需要进行绘制，于是对常用方法进行封装并总结。

1、绘制一个有填充色的圆角矩形​

// 有填充色
/**绘制一个有填充色的圆角矩形 
  *@param ctx:canvas的上下文环境 
  *@param x:左上角x轴坐标 
  *@param y:左上角y轴坐标 
  *@param width:矩形的宽度 
  *@param height:矩形的高度 
  *@param radius:圆的半径 
  *@param fillColor:填充颜色 
**/
export function fillRoundRect(ctx, x, y, width, height, radius, fillColor) {
  if (2 * radius > width || 2 * radius > height) { return false; }
  ctx.save();
  ctx.translate(x, y);
  //绘制圆角矩形的各个边  
  drawRoundRectPath(ctx, width, height, radius);
  ctx.setFillStyle(fillColor || "#fff")
  ctx.fill();
  ctx.restore();
}

// 绘制方法
/**绘制圆角矩形
  *@param ctx:canvas的上下文环境 
  *@param width:矩形的宽度 
  *@param height:矩形的高度 
  *@param radius:圆的半径 
**/
export function drawRoundRectPath(ctx, width, height, radius) {
  ctx.beginPath(0);
  //从右下角顺时针绘制，弧度从0到1/2PI  
  ctx.arc(width - radius, height - radius, radius, 0, Math.PI / 2);
  //矩形下边线  
  ctx.lineTo(radius, height);
  //左下角圆弧，弧度从1/2PI到PI  
  ctx.arc(radius, height - radius, radius, Math.PI / 2, Math.PI);
  //矩形左边线  
  ctx.lineTo(0, radius);
  //左上角圆弧，弧度从PI到3/2PI  
  ctx.arc(radius, radius, radius, Math.PI, Math.PI * 3 / 2);
  //上边线  
  ctx.lineTo(width - radius, 0);
  //右上角圆弧  
  ctx.arc(width - radius, radius, radius, Math.PI * 3 / 2, Math.PI * 2);
  //右边线  
  ctx.lineTo(width, height - radius);
  ctx.closePath();
}

2、绘制一个圆角矩形，边框​

/**该方法用来绘制圆角矩形 
 *@param ctx:canvas的上下文环境 
  *@param x:左上角x轴坐标 
  *@param y:左上角y轴坐标 
  *@param width:矩形的宽度 
  *@param height:矩形的高度 
  *@param radius:圆的半径 
  *@param lineWidth:线条粗细 
  *@param strokeColor:线条颜色 
  **/
export function strokeRoundRect(ctx, x, y, width, height, radius, /*optional*/ lineWidth, /*optional*/ strokeColor) {
  //圆的直径必然要小于矩形的宽高          
  if (2 * radius > width || 2 * radius > height) { return false; }
  ctx.save();
  ctx.translate(x, y);
  //绘制圆角矩形的各个边  
  drawRoundRectPath(ctx, width, height, radius);
  ctx.lineWidth = lineWidth || 1; //若是给定了值就用给定的值否则给予默认值2  
  ctx.strokeStyle = strokeColor;
  ctx.stroke();
  ctx.restore();
}

3、绘制自动换行文字，支持中英文​

 /**canvas绘制文字换行
  *@param ctx:canvas的上下文环境 
  *@param text:文字内容
  *@param x:x坐标 
  *@param y:y坐标 
  *@param maxWidth:最大宽度
  *@param lineHeight:文字行高 
  *@param isEng:是否为英文 
  *@param isJustify:是否需要两端对齐
**/
export function wrapText(ctx, text, x, y, maxWidth=300, lineHeight=20, isEng=false, isJustify=true) {
  if (typeof text != 'string' || typeof x != 'number' || typeof y != 'number') {
    return;
  }

  if (typeof maxWidth == 'undefined') {
    maxWidth = 300;
  }

  // 字符分隔为数组
  const splitStr = isEng? ' ': '';
  const arrText = text.split(splitStr);
  let line = '';
  const splitStrWidth = ctx.measureText(splitStr).width;

  for (let n = 0; n < arrText.length; n++) {
    const testLine = line + arrText[n] + splitStr;
    const metrics = ctx.measureText(testLine);
    const testWidth = metrics.width;
    if (testWidth > maxWidth && n > 0) {

      drawJustifiedText(ctx, line, x, y, maxWidth, false, splitStrWidth, isJustify);
      line = arrText[n]+splitStr;
      y += lineHeight;
    } else {
      line = testLine;
    }
  }
  drawJustifiedText(ctx, line, x, y, maxWidth, true, splitStrWidth, isJustify);
}

4、绘制两端对齐的文字​

 /**canvas绘制两端对齐文字
  *@param ctx:canvas的上下文环境 
  *@param text:文字内容
  *@param x:x坐标 
  *@param y:y坐标 
  *@param maxWidth:最大宽度
  *@param isLastLine:是否为最后一行，最后一行不居中 
  *@param splitStrWidth:分隔符的宽度，中文分隔符是'', 英文分割符是' '
  *@param isJustify:是否需要两端对齐
**/
function drawJustifiedText(ctx, text, x, y, maxWidth, isLastLine, splitStrWidth, isJustify){
  // 最后一行居左
  if(isLastLine||!isJustify) {
    ctx.fillText(text, x, y);
    return;
  }
  const lineText = text.trim();
  const words = lineText.split(' ');
  const numWords = words.length;
  const textWidth = ctx.measureText(lineText).width;
  const spaceWidth = (maxWidth - textWidth) / Math.max(1, numWords - 1)+splitStrWidth;
  for (let i = 0; i < numWords; i++) {
    const word = words[i];
    ctx.fillText(word, x, y);
    x += ctx.measureText(word).width + spaceWidth;
  }
}

5、绘制圆形、圆角图片（头像）​

// h5加载图片，小程序不支持Image
handleGetImage(src) {
  return new Promise((resolve) => {
    const img = new Image();
    img.src = src;
    img.onload=function() {
      resolve(src)
    }
  })
}

// 小程序版本
handleGetImage(src) {
  return new Promise((resolve) => {
    wx.getImageInfo({
      src,
      success: (res) => {
        resolve(res.path)
      },
      fail: (res)=>{
        resolve(false);
      }
    })
  })
}

const path = await this.handleGetImage('xxx');
context.save()
context.beginPath();
// 绘制圆，或者通过arc和lineTo混合绘制圆角框
context.arc(295, 79, 35, 0, Math.PI * 2);
context.closePath()
context.fill();
context.clip(); // 裁剪
context.drawImage(path, 260, 44, 70, 70) // 绘制图片
context.restore()
